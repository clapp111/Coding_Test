### 기록

1. **DFS**
    
    해당 문제를 DFS로 풀게 된다면 핵심은 가지치기이다. DFS를 재귀로 호출 할 것이기 때문에 함수의 시작 부분에서 이전의 최소 깊이보다 더 들어가게 된다면 즉시 해당 루트를 종료한다. 어차피 깊은 곳에서 target을 찾아도 최소 깊이는 갱신되지 않기 때문이다. 
    
2. **BFS**
    
    해당 문제를 BFS로 풀게 된다면 핵심은 큐에 삽입될 원소의 형태이다. 사실 핵심이라고 할 것도 없다. 단순하게 깊이를 나타내기 위한 편의주의적인 방식이다. 큐에 삽입할 때, 해당 노드는 방문했다고 처리하는 것이 더 중요하지만 기본적인 개념이라고 볼 수 있다.
    

### 제출 코드

```cpp
// DFS
#include <string>
#include <vector>

using namespace std;

vector<bool> visited;
int minLevel;

bool isOneAlphabetDiff(const string& a, const string& b) {
    int diffCount = 0;
    for (int i = 0; i < a.size(); ++i) {
        if (a[i] != b[i]) {
            diffCount++;
            if (diffCount > 1) {
                return false;
            }
        }
    }
    return diffCount == 1;
}

void dfs(const string& current, const string& target, const vector<string>& words, int level) {
    if(level >= minLevel) {
        return;
    }
    if (current == target) {
        minLevel = min(minLevel, level);
        return;
    }

    for (int i = 0; i < words.size(); ++i) {
        if (!visited[i] && isOneAlphabetDiff(current, words[i])) {
            visited[i] = true;
            dfs(words[i], target, words, level + 1);
            visited[i] = false;
        }
    }
}

int solution(string begin, string target, vector<string> words) {
    visited = vector<bool>(words.size(), false);
    minLevel = 51;
    dfs(begin, target, words, 0);

    if(minLevel == 51) {
        minLevel = 0;
    }
    return minLevel;
}
```

```cpp
// BFS
#include <string>
#include <vector>
#include <queue>

using namespace std;

bool isOneAlphabetDiff(const string& a, const string& b) {
    int diffCount = 0;
    for (int i = 0; i < a.size(); ++i) {
        if (a[i] != b[i]) {
            diffCount++;
            if (diffCount > 1) {
                return false;
            }
        }
    }
    return diffCount == 1;
}

int bfs(const string& begin, const string& target, const vector<string>& words){
    vector<bool> visited(words.size(), false);
    queue<pair<int, string>> q;

    q.push({0, begin});
    while(!q.empty()){
        auto [level, current] = q.front();
        q.pop();
        if(current == target){
            return level;
        }
        for(int i = 0; i < words.size(); ++i){
            if(!visited[i] && isOneAlphabetDiff(current, words[i])){
                visited[i] = true;
                q.push({level + 1, words[i]});
            }
        }

    }
    return 0;
}

int solution(string begin, string target, vector<string> words) {
    return bfs(begin, target, words);
}
```
